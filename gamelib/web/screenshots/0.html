#include <gamelib/gamelib.h>

int main() {
    VideoModeSetTitle("Simple Example");
    VideoModeSet(640, 480, true);

    while (!key[_esc]) {
        VideoModeEnable2D();
        {
            ColorSet(0.0f, 0.0f, 0.0f, 0.5f);
            DrawFilledRectangle(0, 0, screenWidth, 90);
        }
        VideoModeDisable2D();
        VideoModeFrame();
    }

    return 0;
}
